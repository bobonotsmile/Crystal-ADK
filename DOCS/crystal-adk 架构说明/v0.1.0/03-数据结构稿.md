# crystal-adk v0.1.0 数据结构稿

## `Core/AgentContracts.cs`

```csharp
public sealed class RuntimeMessage
{
    // 消息角色：system / user / assistant / tool
    public string Role { get; set; } = "user";

    // 消息正文
    public string? Content { get; set; }

    // tool 结果消息关联的调用 ID
    public string? ToolCallId { get; set; }

    // tool 结果消息对应的工具名
    public string? Name { get; set; }

    // assistant 请求的工具调用列表
    public List<RuntimeToolCall>? ToolCalls { get; set; }
}

public sealed class RuntimeToolCall
{
    // 单次工具调用 ID
    public string Id { get; set; } = string.Empty;

    // 工具名
    public string Name { get; set; } = string.Empty;

    // 工具参数
    public Dictionary<string, object?> Arguments { get; set; } =
        new(StringComparer.OrdinalIgnoreCase);
}

public sealed class AgentToolDescriptor
{
    // 工具名
    public string Name { get; set; } = string.Empty;

    // 工具描述
    public string Description { get; set; } = string.Empty;

    // 工具参数 schema
    public object ParametersSchema { get; set; } =
        new { type = "object", properties = new { } };
}

public sealed class AgentRunRequest
{
    // 本次运行输入消息
    public List<RuntimeMessage> Messages { get; set; } = new();

    // tool loop 最大轮数
    public int? MaxRounds { get; set; }
}

public sealed class ToolRunRecord
{
    // 对应的调用 ID
    public string ToolCallId { get; set; } = string.Empty;

    // 工具名
    public string ToolName { get; set; } = string.Empty;

    // 调用参数快照
    public Dictionary<string, object?> Arguments { get; set; } =
        new(StringComparer.OrdinalIgnoreCase);

    // 工具执行结果
    public object? Result { get; set; }

    // 工具执行耗时
    public long ElapsedMs { get; set; }
}

public sealed class AgentRunResult
{
    // 最终 assistant 文本
    public string FinalMessage { get; set; } = string.Empty;

    // 运行后完整消息历史
    public List<RuntimeMessage> Messages { get; set; } = new();

    // 运行期间工具记录
    public List<ToolRunRecord> ToolRuns { get; set; } = new();
}
```

public static class AgentEventKinds
{
    // 运行开始
    public const string RunStarted = "run_started";

    // 工具调用开始
    public const string ToolCallStarted = "tool_call_started";

    // 工具调用完成
    public const string ToolCallCompleted = "tool_call_completed";

    // 文本增量
    public const string TextDelta = "text_delta";

    // 最终回答
    public const string FinalAnswer = "final_answer";

    // 运行完成
    public const string RunCompleted = "run_completed";

    // 运行失败
    public const string RunFailed = "run_failed";
}

public sealed class AgentEvent
{
    // 事件种类
    public string Kind { get; init; } = string.Empty;

    // 本次运行 ID
    public string? RunId { get; init; }

    // 工具调用 ID
    public string? ToolCallId { get; init; }

    // 工具名
    public string? ToolName { get; init; }

    // 文本事件内容
    public string? Text { get; init; }

    // 结果对象
    public object? Result { get; init; }

    // 工具参数
    public Dictionary<string, object?>? Arguments { get; init; }

    // 耗时
    public long? ElapsedMs { get; init; }

    // 错误码
    public string? ErrorCode { get; init; }

    // 错误信息
    public string? ErrorMessage { get; init; }
}

## `Core/Providers.cs`

```csharp
public interface IChatProvider
{
    // 非流式返回完整 assistant message
    Task<RuntimeMessage> CreateCompletionAsync(
        IReadOnlyList<RuntimeMessage> messages,
        IReadOnlyList<AgentToolDescriptor> tools,
        CancellationToken cancellationToken);

    // 流式返回统一分片
    IAsyncEnumerable<StreamingChatChunk> StreamCompletionAsync(
        IReadOnlyList<RuntimeMessage> messages,
        IReadOnlyList<AgentToolDescriptor> tools,
        CancellationToken cancellationToken);
}

public sealed class StreamingChatChunk
{
    // 文本增量
    public string? TextDelta { get; init; }

    // 推理文本增量
    public string? ReasoningDelta { get; init; }

    // 当前消息快照；主要用于流式 tool call
    public RuntimeMessage? SnapshotMessage { get; init; }

    // 流式结束原因
    public string? FinishReason { get; init; }

    // 本轮流是否完成
    public bool IsCompleted { get; init; }
}
```
